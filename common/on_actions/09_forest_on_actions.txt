	#Forest Border Wars
	on_startup = {
		effect = {
			MAN = { country_event = { id = FOREST_border_wars.1 hours = 2 } }
			UFC = {
				country_event = { id = UFC_START_EVENT.1 hours = 1 }

	#Forest War 1
	on_capitulation = {
		effect = {
			if = {
				limit = { # The Bears Wins
					ROOT = {
						original_tag = MAN
					
					}
					FROM = {
						original_tag = UFC
					}
				}
				FROM = {
					MAN = { country_event = FOREST_wars.2 } # Lose Event
					MYS = { country_event = { id = FOREST_dipper_kill_multi.1 days = 20} } 
					UFC = {
						country_event = FOREST_wars.1 # Win Event
						IF = { 
							limit = {
								UNI = {
									check_variable = { UNI_UFC_influence > 99 }
								}
							}
							country_event = { id = UNI_intervention.5 days = 1 }
						}
					}
				}
				UNI = { clr_country_flag = UNI_yes_intervene }
			}
			######################################################
			else_if = {
				limit = { # The MANs win
					ROOT = {
						original_tag = UFC
					
					}
					FROM = {
						original_tag = MAN
					}
				}
				FROM = {
					UFC = { country_event = FOREST_wars.2 }  # Lose Event
					MYS = { country_event = { id = FOREST_dipper_kill_man.1 days = 20} } 
					MAN = {
						country_event = FOREST_wars.1 # Win Event
						IF = {
							limit = {
								UNI = {
									check_variable = { UNI_MAN_influence > 99 }
								}
							}
							country_event = { id = UNI_intervention.5 days = 1 }
						}
					}
				}
				UNI = { clr_country_flag = UNI_yes_intervene }
			}
			######################################################
			else_if = {
				limit = { # UNI loses war
					ROOT = {
						original_tag = UNI
					}
				}
				FROM = {
					white_peace = {
						tag = UNI
						message = forest_manos_or_multi_conquers_uni
					}
					annex_country = {
						target = UNI
						transfer_troops = no
					}
				}
			}
			######################################################
			else_if = {
				limit = { # UNI conquers north
					ROOT = {
						OR = {
							original_tag = MAN
							original_tag = UFC
						}
					}
				}
				FROM = {
					white_peace = {
						tag = ROOT
						message = forest_uni_conquers_north
					}
					annex_country = {
						target = ROOT
						transfer_troops = no
					}
				}
			}
			######################################################
		}
