	}

		army = no

		trade = no
	}
	friend =
	{
		army = yes

		submarine = yes
		trade = yes
	}
	neutral =

		army = yes
		navy = yes
		submarine = yes
		trade = yes
	}

		tooltip = disable_canal_tt
	}
		
	icon = 7943
	offset = { 0 -10 0 }
}
